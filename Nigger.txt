        body { margin: 0; display: flex; flex-direction: column; align-items: center; justify-content: center; height: 100vh; background: #222; color: white; font-family: sans-serif; overflow: hidden; }
        canvas { background: #333; border: 4px solid #555; box-shadow: 0 0 20px rgba(0,0,0,0.5); }
        .ui { position: absolute; top: 20px; text-align: center; pointer-events: none; }
        #msg { font-size: 24px; font-weight: bold; margin-top: 10px; color: #ffeb3b; }
    </style>

<body>

    <div class="ui">
        <div>Stage: <span id="stageNum">1</span>/4 | Nyawa: <span id="lives">3</span></div>
        <div id="msg">Lompat (Spasi/Klik) untuk bertahan!</div>
    </div>
    <canvas id="gameCanvas"></canvas>

<script>
const canvas = document.getElementById("gameCanvas");
const ctx = canvas.getContext("2d");

canvas.width = 800;
canvas.height = 400;

// Variabel Game
let stage = 1;
let lives = 3;
let gameActive = true;
let frame = 0;

// Player
const player = {
    x: 100,
    y: 300,
    w: 40,
    h: 40,
    dy: 0,
    jumpForce: 12,
    gravity: 0.6,
    grounded: false,
    color: "#00ffcc"
};

// Rintangan (Obstacles)
let obstacles = [];
const stageConfigs = {
    1: { speed: 5, freq: 100, color: "#4caf50" }, // Mudah
    2: { speed: 7, freq: 80, color: "#2196f3" },  // Sedang
    3: { speed: 10, freq: 60, color: "#ff9800" }, // Cepat
    4: { speed: 13, freq: 40, color: "#f44336" }  // Ekstrim
};

// Kontrol
function jump() {
    if (player.grounded && gameActive) {
        player.dy = -player.jumpForce;
        player.grounded = false;
    }
}

window.addEventListener("keydown", (e) => { if (e.code === "Space") jump(); });
canvas.addEventListener("mousedown", jump);

function spawnObstacle() {
    const config = stageConfigs[stage];
    if (frame % config.freq === 0) {
        let type = Math.random() > 0.5 ? "floor" : "high"; // Rintangan bawah atau melayang
        obstacles.push({
            x: canvas.width,
            y: type === "floor" ? 310 : 230,
            w: 30,
            h: type === "floor" ? 40 : 40,
            color: config.color
        });
    }
}

function update() {
    if (!gameActive) return;

    // Gravitasi & Gerakan
    player.dy += player.gravity;
    player.y += player.dy;

    // Batas Lantai
    if (player.y + player.h > 350) {
        player.y = 350 - player.h;
        player.dy = 0;
        player.grounded = true;
    }

    // Update Obstacles
    spawnObstacle();
    obstacles.forEach((obs, index) => {
        obs.x -= stageConfigs[stage].speed;

        // Collision Detection
        if (player.x < obs.x + obs.w &&
            player.x + player.w > obs.x &&
            player.y < obs.y + obs.h &&
            player.y + player.h > obs.y) {
            
            lives--;
            obstacles.splice(index, 1);
            document.getElementById("lives").innerText = lives;
            
            if (lives <= 0) {
                gameActive = false;
                document.getElementById("msg").innerText = "GAME OVER! Refresh untuk ulang.";
            }
        }

        // Hapus rintangan yang lewat
        if (obs.x + obs.w < 0) obstacles.splice(index, 1);
    });

    // Progres Stage (Setiap 500 frame ganti stage)
    frame++;
    if (frame > 1000 && stage < 4) {
        stage++;
        frame = 0;
        obstacles = [];
        document.getElementById("stageNum").innerText = stage;
        document.getElementById("msg").innerText = "STAGE " + stage + " DIMULAI!";
        setTimeout(() => { if(gameActive) document.getElementById("msg").innerText = ""; }, 2000);
    } else if (stage === 4 && frame > 1500) {
        gameActive = false;
        document.getElementById("msg").innerText = "SELAMAT! Kamu Menang!";
    }
}

function draw() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // Gambar Lantai
    ctx.fillStyle = "#555";
    ctx.fillRect(0, 350, canvas.width, 50);

    // Gambar Player
    ctx.fillStyle = player.color;
    ctx.shadowBlur = 10;
    ctx.shadowColor = player.color;
    ctx.fillRect(player.x, player.y, player.w, player.h);
    ctx.shadowBlur = 0;

    // Gambar Rintangan
    obstacles.forEach(obs => {
        ctx.fillStyle = obs.color;
        ctx.fillRect(obs.x, obs.y, obs.w, obs.h);
    });

    update();
    requestAnimationFrame(draw);
}

draw();
</script>
